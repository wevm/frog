# Frog.frame Response

The response returned from the `.frame` handler.

```tsx twoslash
// @noErrors
import { Frog } from 'frog'
 
export const app = new Frog({ title: 'Frog Frame' })

app.frame('/', (c) => {
  return c.res({ // [!code focus]
    // ... // [!code focus]
  }) // [!code focus]
})
```

## action

- **Type:** `string`

Path of the next frame to perform a `POST` request.

```tsx twoslash
/** @jsxImportSource frog/jsx */
// ---cut---
import { Button, Frog } from 'frog'
 
export const app = new Frog({ title: 'Frog Frame' })
 
app.frame('/', (c) => {
  const { buttonIndex } = c
  return c.res({
    action: '/submit', // [!code focus]
    image: (
      <div style={{ color: 'white', display: 'flex', fontSize: 60 }}>
        Select a frog
      </div>
    ),
    intents: [
      <Button value="apple">Apple</Button>, 
      <Button value="banana">Banana</Button>, 
      <Button value="mango">Mango</Button> 
    ]
  })
})
```

## browserLocation

- **Type:** `string`

Location (URL or path relative to `basePath`) to redirect to when the user
is coming to the page via a browser.

For instance, a user may reach the frame page in their
browser by clicking on the link beneath the frame on Warpcast.
We may want to redirect them to a different page (ie. a mint page, etc)
when they arrive via their browser.

[Read more.](/concepts/browser-redirects)

```tsx twoslash
/** @jsxImportSource frog/jsx */
// ---cut---
import { Button, Frog } from 'frog'
 
export const app = new Frog({ title: 'Frog Frame' })
 
app.frame('/frame/foo', (c) => {
  const { buttonIndex } = c
  return c.res({
    browserLocation: '/foo', // [!code focus]
    image: (
      <div style={{ color: 'white', display: 'flex', fontSize: 60 }}>
        Select a frog
      </div>
    ),
    intents: [
      <Button value="apple">Apple</Button>, 
      <Button value="banana">Banana</Button>, 
      <Button value="mango">Mango</Button> 
    ]
  })
})
```

## headers

- **Type:** `Record<string, string>`

HTTP response headers to set for the frame.

```tsx twoslash
// @noErrors
/** @jsxImportSource frog/jsx */
// ---cut---
import { Button, Frog } from 'frog'
 
export const app = new Frog({ title: 'Frog Frame' })
 
app.frame('/frame/foo', (c) => {
  const { buttonIndex } = c
  return c.res({
    headers: { // [!code focus]
      'cache-control': 'max-age=0', // [!code focus]
    }, // [!code focus]
    image: (
      <div style={{ color: 'white', display: 'flex', fontSize: 60 }}>
        Select a frog
      </div>
    ),
    intents: [
      <Button value="apple">Apple</Button>, 
      <Button value="banana">Banana</Button>, 
      <Button value="mango">Mango</Button> 
    ]
  })
})
```

## image

- **Type:** `string`

The Image to render for the frame. Can either be a JSX element, or URL.

```tsx twoslash
/** @jsxImportSource frog/jsx */
// ---cut---
import { Button, Frog } from 'frog'
 
export const app = new Frog({ title: 'Frog Frame' })
 
app.frame('/', (c) => {
  const { buttonIndex } = c
  return c.res({
    action: '/submit',
    image: ( // [!code focus]
      <div style={{ color: 'white', display: 'flex', fontSize: 60 }}> // [!code focus]
        Select a frog // [!code focus]
      </div> // [!code focus]
    ), // [!code focus]
    intents: [
      <Button value="apple">Apple</Button>, 
      <Button value="banana">Banana</Button>, 
      <Button value="mango">Mango</Button> 
    ]
  })
})
```

```tsx twoslash
// @noErrors
/** @jsxImportSource frog/jsx */
// ---cut---
import { Button, Frog } from 'frog'
 
export const app = new Frog({ title: 'Frog Frame' })
 
app.frame('/', (c) => {

    intents: [
      <Button value="apple">Apple</Button>, 
      <Button value="banana">Banana</Button>, 
      <Button value="mango">Mango</Button> 
    ]
  })
})
```

## imageAspectRatio

- **Type:** `"1.91:1" | "1:1"`

The aspect ratio of the image to render for the frame.

```tsx twoslash
/** @jsxImportSource frog/jsx */
// ---cut---
import { Button, Frog } from 'frog'
 
export const app = new Frog({ title: 'Frog Frame' })
 
app.frame('/', (c) => {
  const { buttonIndex } = c
  return c.res({
    image: ( 
      <div style={{ color: 'white', display: 'flex', fontSize: 60 }}> 
        Select a frog 
      </div> 
    ), 
    imageAspectRatio: '1:1', // [!code focus]
    intents: [
      <Button value="apple">Apple</Button>, 
      <Button value="banana">Banana</Button>, 
      <Button value="mango">Mango</Button> 
    ]
  })
})
```

## imageOptions

- **Type:** `ImageResponseOptions`

[Options for the image](https://vercel.com/docs/functions/og-image-generation/og-image-api) to render for the frame.

```tsx twoslash
/** @jsxImportSource frog/jsx */
// ---cut---
import { Button, Frog } from 'frog'
 
export const app = new Frog({ title: 'Frog Frame' })
 
app.frame('/', (c) => {
  const { buttonIndex } = c
  return c.res({
    image: ( 
      <div style={{ color: 'white', display: 'flex', fontSize: 60 }}> 
        Select a frog 
      </div> 
    ), 
    imageOptions: { // [!code focus]
      height: 600, // [!code focus]
      width: 600, // [!code focus]
    }, // [!code focus]
    intents: [
      <Button value="apple">Apple</Button>, 
      <Button value="banana">Banana</Button>, 
      <Button value="mango">Mango</Button> 
    ]
  })
})
```

:::warning
The `fonts` property is not available on the `imageOptions` passed to `c.res`. If you would like to add custom fonts, either:

1. Add it to `imageOptions.fonts` on the `new Frog({ title: 'Frog Frame' }){:js}` instance:

```tsx twoslash
/** @jsxImportSource frog/jsx */
// ---cut---
import { Frog } from 'frog'
 
export const app = new Frog({
  imageOptions: { fonts: [/* ... */] }, // [!code focus]
  title: 'Frog Frame',
})
```

2. Add it to the route options:

```tsx twoslash
// @noErrors
/** @jsxImportSource frog/jsx */
import { Frog } from 'frog'
 
export const app = new Frog({
  imageOptions: { fonts: [/* ... */] },
  title: 'Frog Frame',
})
// ---cut---
app.frame('/', (c) => {
  return c.res({})
}, {
  fonts: [/* ... */] // [!code focus]
})
```

:::

## intents

A set of intents (ie. buttons, text inputs, etc) to render for the frame
(beneath the image).

```tsx twoslash
/** @jsxImportSource frog/jsx */
// ---cut---
import { Button, Frog } from 'frog'
 
export const app = new Frog({ title: 'Frog Frame' })
 
app.frame('/', (c) => {
  const { buttonIndex } = c
  return c.res({
    image: ( 
      <div style={{ color: 'white', display: 'flex', fontSize: 60 }}> 
        Select a frog 
      </div> 
    ), 
    intents: [ // [!code focus]
      <Button value="apple">Apple</Button>, // [!code focus]
      <Button value="banana">Banana</Button>, // [!code focus]
      <Button value="mango">Mango</Button> // [!code focus]
    ] // [!code focus]
  })
})
```

## ogImage

- **Type:** `string`
- **Default:** The `image` property.

Path or URI of the OG image.

```tsx twoslash

import { Button, Frog } from 'frog'
 
export const app = new Frog({ title: 'Frog Frame' })

      <Button value="mango">Mango</Button> 
    ],
    ogImage: 'https://i.ytimg.com/vi/R3UACX5eULI/maxresdefault.jpg', // [!code focus]
  })
})
```
