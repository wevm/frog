# Columns

Renders children horizontally, with consistent spacing (if specified)
between them. Columns can also have a specific `width` value.

## Usage

:::code-group

```tsx twoslash [Code]
/** @jsxImportSource frog/jsx */
// ---cut---
import { createSystem } from 'frog/ui'

const { Columns, Column } = createSystem()

function Example() {
  return (
    <Columns gap="8" grow> // [!code focus]
      <Column backgroundColor="red" width="1/4" /> // [!code focus]
      <Column backgroundColor="red" width="2/4" /> // [!code focus]
//                                          ^|
      <Column backgroundColor="red" width="1/4" /> // [!code focus]
    </Columns> // [!code focus]
  )
}
```

<div data-title="Preview">
  <div style={{ display: 'flex', flexDirection: 'row', gap: 8, height: '100px' }}>
    <div style={{ backgroundColor: 'red', flex: 1, height: '100%' }} />
    <div style={{ backgroundColor: 'red', flex: 2, height: '100%' }} />
    <div style={{ backgroundColor: 'red', flex: 1, height: '100%' }} />
  </div>
</div>

:::

## `Columns` Properties

### `alignHorizontal`

Horizontally aligns the contents.

:::code-group

```tsx twoslash [Code]
/** @jsxImportSource frog/jsx */
import { createSystem } from 'frog/ui'

const { Columns, Column } = createSystem()

function Example() {
  return (
// ---cut---
<Columns 
  alignHorizontal="center" // [!code focus]
//                 ^| 
  grow
>
// ---cut-after---
      <Column backgroundColor="red" width="1/4" /> // [!code focus]
      <Column backgroundColor="red" width="2/4" /> // [!code focus]
      <Column backgroundColor="red" width="1/4" /> // [!code focus]
    </Columns> // [!code focus]
  )
}
```

<div data-title="Preview">
  <div style={{ justifyContent: 'center', display: 'flex', flexDirection: 'row', gap: 8, height: '100px' }}>
    <div style={{ backgroundColor: 'red', flex: '0.25', height: '100%' }} />
    <div style={{ backgroundColor: 'red', flex: '0.25', height: '100%' }} />
  </div>
</div>

:::

## `Column` Properties

### `width`

Sets the width span of the column (in fractions).

:::code-group

```tsx twoslash [Code]
/** @jsxImportSource frog/jsx */
// ---cut---
import { createSystem } from 'frog/ui'

const { Columns, Column } = createSystem()

function Example() {
  return (
    <Columns gap="8" grow> 
      <Column backgroundColor="red" width="1/4" /> 
      <Column backgroundColor="red" width="2/4" /> // [!code focus]
//                                          ^|
      <Column backgroundColor="red" width="1/4" /> 
    </Columns> 
  )
}
```

<div data-title="Preview">
  <div style={{ display: 'flex', flexDirection: 'row', gap: 8, height: '100px' }}>
    <div style={{ backgroundColor: 'gray', flex: 1, height: '100%' }} />
    <div style={{ backgroundColor: 'red', flex: 2, height: '100%' }} />
    <div style={{ backgroundColor: 'gray', flex: 1, height: '100%' }} />
  </div>
</div>

:::
