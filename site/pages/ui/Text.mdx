import * as React from 'react'

# Text

Renders a text node.

## Usage

:::code-group

```tsx twoslash [Code]
/** @jsxImportSource frog/jsx */
// ---cut---
import { createSystem } from 'frog/ui'

const { Text } = createSystem()

function Example() {
  return <Text>Hello world</Text>
}
```

<div data-title="Preview">
  <div>Hello world</div>
</div>

:::

## `Text` Properties

### `align`

Sets the horizontal alignment of the text.

:::code-group

```tsx twoslash [Code]

// ---cut---

  return <Text align="right">Hello world</Text>
//                    ^|
}



```

<div data-title="Preview">
  <div style={{ textAlign: 'right' }}>Hello world</div>
</div>

:::
