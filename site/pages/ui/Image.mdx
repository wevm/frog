import * as React from 'react'

# Image

Renders an image.

## Usage

:::code-group
```tsx twoslash [Code]
/** @jsxImportSource frog/jsx */
// ---cut---
import { createSystem } from 'frog/ui'

const { Image } = createSystem()

function Example() {
  return <Image src="/frog.png" />
}
```

<div data-title="Preview">
  <img src="/frog.png" />
</div>
:::

## `Image` Properties

### `borderRadius`

Sets the border radius of the image.

Also, works with `border{Bottom|Top}{Bottom|Left|Right|Top}Radius`.

:::code-group
```tsx twoslash [Code]
/** @jsxImportSource frog/jsx */
import { createSystem } from 'frog/ui'

const { Image } = createSystem()

function Example() {
  return (
// ---cut---
<Image
  borderRadius="48" // [!code focus]
//              ^|
  src="/frog.png"
/>
// ---cut-after---
  )
}



```

<div data-title="Preview">
  <img src="/frog.png" style={{ borderRadius: '48px' }} />
</div>
:::
