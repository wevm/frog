import * as React from 'react'

# Heading

Renders a heading text node.

## Usage

:::code-group

```tsx twoslash [Code]
/** @jsxImportSource frog/jsx */
// ---cut---
import { createSystem } from 'frog/ui'

const { Heading } = createSystem()

function Example() {
  return <Heading>Hello world</Heading>
}
```

<div data-title="Preview">
  <div style={{ fontWeight: '600', fontSize: 32 }}>Hello world</div>
</div>

:::

## `Heading` Properties

### `align`

Sets the horizontal alignment of the text.

:::code-group

```tsx twoslash [Code]
/** @jsxImportSource frog/jsx */
import { createSystem } from 'frog/ui'

const { Heading } = createSystem()

// ---cut---
function Example() {
  return <Heading align="right">Hello world</Heading>
//                       ^|
}



```

<div data-title="Preview">
  <div style={{ fontWeight: '600', fontSize: 32, textAlign: 'right' }}>Hello world</div>
</div>

:::

### `color`

Sets the color of the text.

Values are mapped to the [`colors` property](/ui/ui-system#colors) on the UI System Variables.

:::code-group

```tsx twoslash [Code]
/** @jsxImportSource frog/jsx */
import { createSystem } from 'frog/ui'

const { Heading } = createSystem()

// ---cut---
function Example() {
  return <Heading color="red">Hello world</Heading>
//                        ^|
}






```

<div data-title="Preview">
  <div style={{ color: 'red', fontWeight: '600', fontSize: 32 }}>Hello world</div>
</div>

:::

### `font`

Sets the font family of the text.

Values are mapped to the [`fonts` property](/ui/ui-system#fonts) on the UI System Variables.

```tsx twoslash [Code]
// @noErrors

import { createSystem } from 'frog/ui'

const { Heading } = createSystem()

// ---cut---
function Example() {
  return <Heading font="merriweather">Hello world</Heading>
}
```

### `tracking`

Sets the horizontal spacing behavior between text characters.

Values are mapped to the [`units` property](/ui/ui-system#units) on the UI System Variables.

:::code-group

```tsx twoslash [Code]

import { createSystem } from 'frog/ui'

const { Heading } = createSystem()

// ---cut---
function Example() {
  return <Heading tracking="-2">Hello world</Heading>
//                          ^|
}





```

<div data-title="Preview">
  <div style={{ fontWeight: '600', fontSize: 32, letterSpacing: '-2px' }}>Hello world</div>
</div>

:::

### `wrap`

Defines how the text should be wrapped.

:::code-group

```tsx twoslash [Code]
//                      ^|
}
```
